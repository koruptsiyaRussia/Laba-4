max_ratio = 0  # Инициализация переменной для хранения максимального отношения
number = 0  # Инициализация переменной для хранения числа, соответствующего максимальному отношению

for i in range(100, 1000):  # Цикл по всем трехзначным числам
    digit_sum = sum(int(digit) for digit in str(i))  # Вычисление суммы цифр текущего числа
    if digit_sum != 0:  # Проверка, что сумма цифр не равна нулю (чтобы избежать деления на ноль)
        ratio = i / digit_sum  # Вычисление отношения числа к сумме его цифр
        if ratio > max_ratio:  # Проверка, является ли текущее отношение большим, чем ранее найденное максимальное
            max_ratio = ratio  # Обновление максимального отношения
            number = i  # Обновление числа, соответствующего максимальному отношению

print(f"Наибольшее значение отношения трехзначного числа к сумме его цифр: {max_ratio}, число: {number}")  # Вывод результата
